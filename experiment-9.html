Experiment-9: AngularJS Form with CSS and Events

This experiment focuses on creating a form in AngularJS and styling it with CSS, including event handling.

index.html:

<!DOCTYPE html>
<html ng-app="formApp">
<head>
  <meta charset="UTF-8">
  <title>AngularJS Form with CSS and Events</title>
  <script src="https://ajax.googleapis.com/ajax/libs/angularjs/1.8.3/angular.min.js"></script>
  <script src="app.js"></script>
  <style>
    body {
      font-family: Arial, sans-serif;
      padding: 20px;
      background-color: #f2f2f2;
      border-radius: 10px;
      padding: 20px;
      width: 300px;
    }
    input.ng-valid {
      border: 2px solid green;
    }
    input.ng-invalid {
      border: 2px solid red;
    }
    .highlight {
      background-color: yellow;
    }
    button {
      background-color: #4CAF50;
      color: white;
      padding: 10px 15px;
      border: none;
      border-radius: 5px;
    }
    h3 {
      color: darkblue;
    }
  </style>
</head>
<body ng-controller="formCtrl">
  <div class="form-box" ng-mouseover="highlight=true" ng-mouseleave="highlight=false">
    <h3>Student Registration</h3>
    <form name="studentForm" novalidate>
      Name: <br><input type="text" ng-model="student.name" required><br><br>
      Email: <br><input type="email" ng-model="student.email" required><br><br>
      Course: <br>
      <select ng-model="student.course">
        <option value="AngularJS">AngularJS</option>
        <option value="NodeJS">NodeJS</option>
        <option value="MongoDB">MongoDB</option>
      </select><br><br>
      <button type="button" ng-click="submitForm()">Submit</button>
    </form>
    <div ng-show="submitted">
      <h4>Form Submitted:</h4>
      Name: {{student.name}}<br>
      Email: {{student.email}}<br>
      Course: {{student.course}}<br>
    </div>
  </div>
</body>
</html>

app.js:

var app = angular.module(\'formApp\', []);
app.controller(\'formCtrl\', function($scope) {
  $scope.student = {
    name: \'\',
    email: \'\',
    course: \'AngularJS\'
  };
  $scope.submitted = false;

  $scope.submitForm = function() {
    if ($scope.student.name && $scope.student.email) {
      $scope.submitted = true;
    } else {
      alert("Please fill all required fields correctly!");
    }
  };
});
