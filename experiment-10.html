Experiment-10: React JS Voting Application

This experiment involves building a simple voting application using React JS.

App.js:

import React, { useState } from \'react\';

function App() {
  const [candidates, setCandidates] = useState([
    { name: \'Alice\', votes: 0 },
    { name: \'Bob\', votes: 0 },
    { name: \'Charlie\', votes: 0 }
  ]);

  const vote = (index) => {
    const updated = [...candidates];
    updated[index].votes += 1;
    setCandidates(updated);
  };

  const resetVotes = () => {
    const reset = candidates.map(c => ({ ...c, votes: 0 }));
    setCandidates(reset);
  };

  return (
    <div style={styles.container}>
      {candidates.map((candidate, index) => (
        <div key={index} style={styles.card}>
          <span style={styles.name}>{candidate.name}</span>
          <span style={styles.votes}>{candidate.votes}</span>
          <button style={styles.button} onClick={() => vote(index)}>Vote</button>
        </div>
      ))}
      <button style={styles.resetButton} onClick={resetVotes}>Reset</button>
    </div>
  );
}

const styles = {
  container: {
    fontFamily: \'Arial\',
    padding: 30,
    textAlign: \'center\'
  },
  card: {
    margin: 15,
    padding: 15,
    border: \'1px solid #ccc\',
    borderRadius: 10,
    display: \'inline-block\',
    width: 256,
    backgroundColor: \'#f9f9f9\'
  },
  name: {
    fontSize: 20,
    fontWeight: \'bold\',
    color: \'#333\'
  },
  votes: {
    display: \'block\',
    margin: \'10px 0\'
  },
  button: {
    backgroundColor: \'#4CAF50\',
    color: \'white\',
    padding: \'8px 16px\',
    border: \'none\',
    borderRadius: 5,
    cursor: \'pointer\'
  },
  resetButton: {
    marginTop: 30,
    backgroundColor: \'#f44336\',
    color: \'white\',
    padding: \'10px 20px\',
    border: \'none\',
    borderRadius: 5,
    cursor: \'pointer\'
  }
};

export default App;
