Experiment-3: Node.js File Operations

This experiment covers reading from, writing to, and deleting files using Node.js.

file1.txt:

Hello this is FSD Lab

readfile.js:

//READ FILE
var http = require(\'http\');
var fs = require(\'fs\');
http.createServer(function (req, res) {
fs.readFile(\'file1.txt\', function(err, data) {
res.writeHead(200, {\'Content-Type\': \'text/html\'});
res.write(data);
return res.end();
});
}).listen(3000);

appendfile.js:

//APPEND FILE ASYNCHRONOUS CALL
var fs = require(\'fs\');
fs.appendFile(\'file1.txt\', \'FULL STACK DEVELOPMENT\', function (err) {
if (err) throw err;
console.log(\'Saved!\');
});

delfile.js:

//DELETE FILE
var fs = require(\'fs\');
fs.unlink(\'file1.txt\', function (err) {
if (err) throw err;
});
console.log(\'File deleted!\');
