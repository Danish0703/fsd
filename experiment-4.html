Experiment-4: Reading Form Data from Query String

This experiment demonstrates how to read form data submitted via a query string in Node.js.

dtime.html:

<!DOCTYPE html>
<html>
<body>
<form action="http://localhost:3003">
Year: <input type="text" name="year"><br>
Month: <input type="text" name="month"><br>
<input type="submit">
</form>
</body>
</html>

readform.js:

var http = require(\'http\');
var url = require(\'url\');
http.createServer(function (req, res) {
res.writeHead(200, {\'Content-Type\': \'text/html\'});
var q = url.parse(req.url, true).query;
console.log(q);
var txt = q.year + " " + q.month;
res.end(txt);
}).listen(3003);
